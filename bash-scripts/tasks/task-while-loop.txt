implements console functionality - infinite loop that reads user input on each iteration
supports commands:
ls [dir] - lists the contents of specified directory
pwd - shows the path to current directory
hi - outputs "Hello <name of the current user>" 
exit - ends the script
- run your script and test implemented commands

#!/bin/bash

# while [ true ];
while : ;
do
  read -p "Enter command (exit to stop entering) : " input

  if [[ "$input" == "exit" ]];
  then
    echo "See you later"
    break
  elif [[ "$input" == "ls" ]];
  then
    ls
  elif [[ "$input" == "hi" ]];
  then
    echo "Hello you"
  elif [[ "$input" == "pwd" ]];
  then
    pwd
  else
    echo "Command not supported"
  fi
done

#!/bin/bash

while true; do
  read -p "> " cmd arg

  case "$cmd" in
    ls)
      if [[ -z "$arg" ]]; then
        ls
      else
        ls "$arg"
      fi
      ;;
    pwd)
      pwd
      ;;
    hi)
      echo "Hello $USER"
      ;;
    exit)
      echo "Goodbye!"
      break
      ;;
    *)
      echo "Unknown command: $cmd"
      ;;
  esac
done
